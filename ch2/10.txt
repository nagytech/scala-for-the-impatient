def fxn(x: Double, n: Int): Double = {
  
  if (n == 0)
  {
    1
  }
  else if (n > 0)
  {
    if (n % 2 == 0) {
      fxn(fxn(x, n / 2), 2)
    } else {
      x * fxn(x, n - 1)
    }
  }
  else 
  {
    1 / fxn(x, -n)
  }

}

println(fxn(5,4))  // 5^4/2, recurse to 5^2, recurse to 5^1, return as 5*5^1-1 = 5*1 = 5
println(fxn(5,3))  // 5*5^3-1 = 5*5^2, recurse as above to end up at 5*5 = 25
println(fxn(5,0))  // 1
println(fxn(5,-1)) // 1 / (5^-(-1) recurse to 5*5^1 = 5), overall 1/5 = 0.2
